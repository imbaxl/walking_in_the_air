$$\begin{align}
&(C)'=0 &\quad & \mathrm{d}(C)=0\cdot \mathrm{d}x=0 \\
&(x^a)'=ax^{n-1} &\quad & \mathrm{d}(x^a)=ax^{a-1}\mathrm{d}x \\
&(\sin x)'=\cos x &\quad & \mathrm{d}(\sin x)=\cos \mathrm{d}x \\
&(\cos x)'=-\sin x &\quad & \mathrm{d}(\cos x)=-\sin x\mathrm{d}x  \\
&(\tan x)'=\sec^2x &\quad& \mathrm{d}(\tan x)=\sec^2x\mathrm{d}x \\
&(\cot x)'=-\csc^2x &\quad&  \mathrm{d}(\cot x)=-\csc^2x\mathrm{d}x \\
&(\sec x)'=\tan x\sec x &\quad& \mathrm{d}(\sec x)=\tan x\sec x\mathrm{d}x \\
&(\csc x)'=-\cot x\csc x &\quad& \mathrm{d}(\csc x)=-\cot x\csc x\mathrm{d}x \\
&(\arcsin x)' =\dfrac{1}{\sqrt{ 1-x^2 }}&\quad& \mathrm{d}(\arcsin x)=\dfrac{\mathrm{d}x}{\sqrt{ 1-x^2 }}   \\
&(\arccos x)'=-\dfrac{1}{\sqrt{ 1-x^2 }} &\quad& \mathrm{d}(\arccos x)=-\dfrac{\mathrm{d}x}{\sqrt{ 1-x^2 }} \\
&(\arctan x)'=\dfrac{1}{1+x^2} &\quad& \mathrm{d}(\arctan x)=\dfrac{\mathrm{d}x}{1+x^2} \\
&(\mathrm{arccot}\, x)'=-\dfrac{1}{1+x^2} &\quad& \mathrm{d}(\mathrm{arccot}\,x)=-\dfrac{\mathrm{d}x}{1+x^2}  \\
&(a^x)'=a^x \ln a &\quad& \mathrm{d}(a^x)=a^x \ln a \mathrm{d}x \\
&(e^x)'=e^x &\quad& \mathrm{d}(e^x)=e^x\mathrm{d}x \\
&(\log_{a}x)'=\dfrac{1}{\ln a}\dfrac{1}{x} &\quad& \mathrm{d}(\log_{a}x)=\dfrac{1}{\ln a}\dfrac{\mathrm{d}x}{x} \\
&(\mathrm{sh}\,x)'=\mathrm{ch}\,x &\quad& \mathrm{d}(\mathrm{sh}\,x)=\mathrm{ch}\,x\mathrm{d}x \\
&\dots &\quad& \dots
\end{align}$$
